243inline void
244remove_surrounds(std::string& s)
245{
246 std::size_t len = s.length();
247 while (len > 1) {
248 // If the first character is alpha-numeric we are done.
249 char first = s[0];
250 if (isalnum(first)) return;
251
252 // First character is not alpha-numeric.
253 // Grab the last character & check for a match.
254 char last = s[len - 1];
255 bool match = false;
256
257 // Handle cases [text], {text}, <text>, and (text) and then all others
258 switch (first) {
259 case '(':
260 if (last == ')') match = true;
261 break;
262 case '[':
263 if (last == ']') match = true;
264 break;
265 case '{':
266 if (last == '}') match = true;
267 break;
268 case '<':
269 if (last == '>') match = true;
270 break;
271 default:
272 if (last == first) match = true;
273 break;
274 }
275
276 if (match) {
277 // Shrink the string and continue
278 s = s.substr(1, len - 2);
279 len -= 2;
280 }
281 else {
282 // No match => no surround so we can exit.
283 return;
284 }
285 }
286}

630template<std::input_iterator InputIt, std::forward_iterator ForwardIt, typename BinaryFunc>
631constexpr void
632for_each_token(InputIt ib, InputIt ie, ForwardIt db, ForwardIt de, BinaryFunc function)
633{
634 while (ib != ie) {
635 const auto x = std::find_first_of(ib, ie, db, de); // Find a token in the input text
636 function(ib, x); // Call the user supplied function on the token
637 if (x == ie) break; // Stop if we hit the end of the input text
638 ib = std::next(x); // Otherwise go again past that token we just found
639 }
640}

657template<typename Container_t>
658constexpr void
659tokenize(std::string_view input, Container_t& output, std::string_view delimiters = "\t,;: ", bool skip = true)
660{
661 auto ib = cbegin(input);
662 auto ie = cend(input);
663 auto db = cbegin(delimiters);
664 auto de = cend(delimiters);
665
666 for_each_token(ib, ie, db, de, [&output, &skip](auto tb, auto te) {
667 if (tb != te || !skip) { output.emplace_back(tb, te); }
668 });
669}

701template<typename Iter, typename Traits, typename CharT, typename UnaryFunction>
702std::basic_string<CharT>
703regex_replace(Iter ib, Iter ie, const std::basic_regex<CharT, Traits>& re, UnaryFunction f)
704{
705 std::basic_string<CharT> s;
706
707 using diff_t = typename std::match_results<Iter>::difference_type;
708 diff_t match_pos_old = 0;
709 auto end_last_match = ib;
710
711 auto callback = [&](const std::match_results<Iter>& match) {
712 auto match_pos = match.position(0);
713 auto diff = match_pos - match_pos_old;
714
715 auto start_match = end_last_match;
716 std::advance(start_match, diff);
717
718 s.append(end_last_match, start_match);
719 s.append(f(match));
720
721 auto match_len = match.length(0);
722 match_pos_old = match_pos + match_len;
723 end_last_match = start_match;
724 std::advance(end_last_match, match_len);
725 };
726
727 std::regex_iterator<Iter> begin(ib, ie, re), end;
728 std::for_each(begin, end, callback);
729 s.append(end_last_match, ie);
730 return s;
731}
